053/**
054 * Convert binary execution data to XML report including dependencies. <br>
055 * This mojo will convert the binary jacoco execution data into the same format
056 * as the XML report. But it will take all module dependencies into
057 * consideration. This may be helpful when tools actually require XML data, but
058 * tests in a module are responsible for testing classes in other modules.
059 */
060@Mojo(defaultPhase = VERIFY, name = "xml", requiresProject = true, inheritByDefault = true, requiresDependencyResolution = ResolutionScope.TEST)
061public class XmlMojo extends AbstractMojo {
062
063        /**
064         * Allows to skip the execution
065         */
066        @Parameter(property = "jacoco.extras.skip", defaultValue = "false")
067        private boolean skip;
068
069        /**
070         * The jacoco execution data <br>
071         * If this file doesn't exist, execution of this plugin will be skipped
072         */
073        @Parameter(property = "jacoco.extras.execFile", defaultValue = "${project.build.directory}/jacoco.exec", required = true)
074        private File execFile;
075
076        /**
077         * The output XML file
078         */
079        @Parameter(property = "jacoco.extras.xmlFile", defaultValue = "${project.build.directory}/jacoco.xml", required = true)
080        private File xmlFile;
081
082        /**
083         * The encoding of the source files
084         */
085        @Parameter(property = "project.build.sourceEncoding", defaultValue = "UTF-8")
086        private String sourceEncoding;
087
088        /**
089         * Include patterns. The default is to include everything.
090         */
091        @Parameter
092        private List<String> includes;
093
094        /**
095         * Exclude patterns. The default is to exclude nothing.
096         */
097        @Parameter
098        private List<String> excludes;
099
100        @Parameter(property = "project", readonly = true)
101        private MavenProject project;
102
103        @Parameter(property = "reactorProjects", readonly = true)
104        private List<MavenProject> reactorProjects;
105
106        /**
107         * If the XML file should be pretty printed.
108         *
109         * @since 0.1.2
110         */
111        @Parameter(property = "jacoco.extras.pretty", defaultValue = "true")
112        private boolean pretty = true;
113
114        /**
115         * When pretty printing, if the original file should be deleted.
116         *
117         * @since 0.1.2
118         */
119        @Parameter(property = "jacoco.extras.deleteRaw", defaultValue = "true")
120        private final boolean deleteRaw = true;
121
122        public void setPretty(final boolean pretty) {
123                this.pretty = pretty;
124        }
125
126        @Override
127        public void execute() throws MojoExecutionException, MojoFailureException {
128                if (this.skip) {
129                        return;
130                }
131
132                if (!this.execFile.isFile()) {
133                        getLog().debug("Not running. No execution data found.");
134                        return;
135                }
136
137                try {
138                        this.xmlFile.getParentFile().mkdirs();
139
140                        final ReportSupport report = new ReportSupport(getLog());
141                        report.loadExecutionData(this.execFile);
142                        report.addXmlFormatter(this.xmlFile, "UTF-8");
143
144                        final IReportVisitor visitor = report.initRootVisitor();
145                        final IReportGroupVisitor group = visitor.visitGroup("XML");
146
147                        processProject(report, group, this.project);
148
149                        for (final MavenProject dependency : findDependencies(SCOPE_COMPILE, SCOPE_RUNTIME, SCOPE_PROVIDED,
150                                        SCOPE_TEST)) {
151                                processProject(report, group, dependency);
152                        }
153
154                        visitor.visitEnd();
155
156                        if (this.pretty) {
157                                makePretty();
158                        }
159
160                } catch (final IOException e) {
161                        throw new MojoExecutionException("Failed to convert to XML", e);
162                }
163        }
164
165        private void processProject(final ReportSupport report, final IReportGroupVisitor group, final MavenProject project)
166                        throws IOException {
167                report.processProject(group, project.getArtifactId(), project, this.includes, this.excludes,
168                                this.sourceEncoding);
169        }
170
171        private void makePretty() throws MojoExecutionException {
172                try {
173                        final TransformerFactory tf = TransformerFactory.newInstance();
174
175                        final Transformer transformer = tf.newTransformer();
176                        transformer.setOutputProperty(OutputKeys.INDENT, "yes");
177                        transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
178                        transformer.setOutputProperty(OutputKeys.ENCODING, this.sourceEncoding);
179                        transformer.setOutputProperty(OutputKeys.DOCTYPE_PUBLIC, "-//JACOCO//DTD Report 1.0//EN");
180                        transformer.setOutputProperty(OutputKeys.DOCTYPE_SYSTEM, "report.dtd");
181
182                        final Path tmp = this.xmlFile.toPath().getParent()
183                                        .resolve("raw." + this.xmlFile.toPath().getFileName().toString());
184
185                        Files.move(this.xmlFile.toPath(), tmp, REPLACE_EXISTING);
186
187                        try (final InputStream in = new BufferedInputStream(Files.newInputStream(tmp));
188                                        final OutputStream out = new BufferedOutputStream(Files.newOutputStream(this.xmlFile.toPath()));) {
189
190                                final DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
191                                dbf.setValidating(false);
192                                dbf.setExpandEntityReferences(false);
193                                dbf.setFeature("http://apache.org/xml/features/nonvalidating/load-external-dtd", false);
194                                final Document doc = dbf.newDocumentBuilder().parse(in);
195
196                                final DOMSource source = new DOMSource(doc);
197                                final StreamResult result = new StreamResult(out);
198
199                                transformer.transform(source, result);
200                        }
201
202                        // we only delete this if we succeeded
203                        if (this.deleteRaw) {
204                                Files.deleteIfExists(tmp);
205                        }
206
207                } catch (final Exception e) {
208                        throw new MojoExecutionException("Failed to pretty print XML file", e);
209                }
210
211        }
212
213        private List<MavenProject> findDependencies(final String... scopes) {
214                final List<MavenProject> result = new LinkedList<>();
215                final Set<String> scopeList = new HashSet<>(Arrays.asList(scopes));
216                for (final Dependency dependency : this.project.getDependencies()) {
217                        if (scopeList.contains(dependency.getScope())) {
218                                getLog().info("Adding dependency - " + dependency.toString());
219                                final MavenProject project = findProjectFromReactor(dependency);
220                                if (project != null) {
221                                        getLog().info("  -> Unable to find in reactor");
222                                        result.add(project);
223                                }
224                        }
225                }
226                return result;
227        }
228
229        private MavenProject findProjectFromReactor(final Dependency d) {
230                for (final MavenProject p : this.reactorProjects) {
231                        if (p.getGroupId().equals(d.getGroupId()) && p.getArtifactId().equals(d.getArtifactId())
232                                        && p.getVersion().equals(d.getVersion())) {
233                                return p;
234                        }
235                }
236                return null;
237        }
