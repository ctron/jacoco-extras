011package de.dentrassi.maven.jacoco;
012
013import static org.apache.maven.plugins.annotations.LifecyclePhase.VERIFY;
014
015import java.io.File;
016import java.io.IOException;
017import java.util.ArrayList;
018import java.util.Arrays;
019import java.util.List;
020
021import org.apache.maven.artifact.Artifact;
022import org.apache.maven.model.Dependency;
023import org.apache.maven.plugin.AbstractMojo;
024import org.apache.maven.plugin.MojoExecutionException;
025import org.apache.maven.plugin.MojoFailureException;
026import org.apache.maven.plugins.annotations.Mojo;
027import org.apache.maven.plugins.annotations.Parameter;
028import org.apache.maven.project.MavenProject;
029import org.jacoco.report.IReportGroupVisitor;
030import org.jacoco.report.IReportVisitor;
031
032@Mojo(defaultPhase = VERIFY, name = "xml", requiresProject = true, inheritByDefault = true)
033public class XmlMojo extends AbstractMojo {
034
035    /**
036     * Allows to skip the execution
037     */
038    @Parameter(property = "jacoco.extras.skip", defaultValue = "false")
039    private boolean skip;
040
041    /**
042     * The jacoco execution data <br>
043     * If this file doesn't exist, execution of this plugin will be skipped
044     */
045    @Parameter(property = "jacoco.extras.execFile", defaultValue = "${project.build.directory}/jacoco.exec", required = true)
046    private File execFile;
047
048    /**
049     * The output XML file
050     */
051    @Parameter(property = "jacoco.extras.xmlFile", defaultValue = "${project.build.directory}/jacoco.xml", required = true)
052    private File xmlFile;
053
054    /**
055     * The encoding of the source files
056     */
057    @Parameter(property = "project.build.sourceEncoding", defaultValue = "UTF-8")
058    private String sourceEncoding;
059
060    /**
061     * Include patterns. The default is to include everything.
062     */
063    @Parameter
064    private List<String> includes;
065
066    /**
067     * Exclude patterns. The default is to exclude nothing.
068     */
069    @Parameter
070    private List<String> excludes;
071
072    @Parameter(property = "project", readonly = true)
073    private MavenProject project;
074
075    @Parameter(property = "reactorProjects", readonly = true)
076    private List<MavenProject> reactorProjects;
077
078    @Override
079    public void execute() throws MojoExecutionException, MojoFailureException {
080        if (this.skip) {
081            return;
082        }
083
084        if (!this.execFile.isFile()) {
085            getLog().debug("Not running. No execution data found.");
086            return;
087        }
088
089        try {
090            this.xmlFile.getParentFile().mkdirs();
091
092            final ReportSupport report = new ReportSupport(getLog());
093            report.loadExecutionData(this.execFile);
094            report.addXmlFormatter(this.xmlFile, "UTF-8");
095
096            final IReportVisitor visitor = report.initRootVisitor();
097            final IReportGroupVisitor group = visitor.visitGroup("XML");
098
099            for (final MavenProject dependency : findDependencies(Artifact.SCOPE_COMPILE, Artifact.SCOPE_RUNTIME)) {
100                report.processProject(group, dependency.getArtifactId(), dependency, this.includes, this.excludes,
101                        this.sourceEncoding);
102            }
103
104            visitor.visitEnd();
105
106        } catch (final IOException e) {
107            throw new MojoExecutionException("Failed to convert to XML", e);
108        }
109    }
110
111    private List<MavenProject> findDependencies(final String... scopes) {
112        final List<MavenProject> result = new ArrayList<>();
113        final List<String> scopeList = Arrays.asList(scopes);
114        for (final Object dependencyObject : this.project.getDependencies()) {
115            final Dependency dependency = (Dependency) dependencyObject;
116            if (scopeList.contains(dependency.getScope())) {
117                final MavenProject project = findProjectFromReactor(dependency);
118                if (project != null) {
119                    result.add(project);
120                }
121            }
122        }
123        return result;
124    }
125
126    private MavenProject findProjectFromReactor(final Dependency d) {
127        for (final MavenProject p : this.reactorProjects) {
128            if (p.getGroupId().equals(d.getGroupId()) && p.getArtifactId().equals(d.getArtifactId())
129                    && p.getVersion().equals(d.getVersion())) {
130                return p;
131            }
132        }
133        return null;
134    }
135
136}
