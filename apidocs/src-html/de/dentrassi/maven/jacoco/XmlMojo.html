052/**
053 * Convert binary execution data to XML report including dependencies. <br>
054 * This mojo will convert the binary jacoco execution data into the same format
055 * as the XML report. But it will take all module dependencies into
056 * consideration. This may be helpful when tools actually require XML data, but
057 * tests in a module are responsible for testing classes in other modules.
058 */
059@Mojo(defaultPhase = VERIFY, name = "xml", requiresProject = true, inheritByDefault = true, requiresDependencyResolution = ResolutionScope.COMPILE_PLUS_RUNTIME)
060public class XmlMojo extends AbstractMojo {
061
062    /**
063     * Allows to skip the execution
064     */
065    @Parameter(property = "jacoco.extras.skip", defaultValue = "false")
066    private boolean skip;
067
068    /**
069     * The jacoco execution data <br>
070     * If this file doesn't exist, execution of this plugin will be skipped
071     */
072    @Parameter(property = "jacoco.extras.execFile", defaultValue = "${project.build.directory}/jacoco.exec", required = true)
073    private File execFile;
074
075    /**
076     * The output XML file
077     */
078    @Parameter(property = "jacoco.extras.xmlFile", defaultValue = "${project.build.directory}/jacoco.xml", required = true)
079    private File xmlFile;
080
081    /**
082     * The encoding of the source files
083     */
084    @Parameter(property = "project.build.sourceEncoding", defaultValue = "UTF-8")
085    private String sourceEncoding;
086
087    /**
088     * Include patterns. The default is to include everything.
089     */
090    @Parameter
091    private List<String> includes;
092
093    /**
094     * Exclude patterns. The default is to exclude nothing.
095     */
096    @Parameter
097    private List<String> excludes;
098
099    @Parameter(property = "project", readonly = true)
100    private MavenProject project;
101
102    @Parameter(property = "reactorProjects", readonly = true)
103    private List<MavenProject> reactorProjects;
104
105    /**
106     * If the XML file should be pretty printed.
107     *
108     * @since 0.1.2
109     */
110    @Parameter(property = "jacoco.extras.pretty", defaultValue = "true")
111    private boolean pretty = true;
112
113    /**
114     * When pretty printing, if the original file should be deleted.
115     *
116     * @since 0.1.2
117     */
118    @Parameter(property = "jacoco.extras.deleteRaw", defaultValue = "true")
119    private final boolean deleteRaw = true;
120
121    public void setPretty(final boolean pretty) {
122        this.pretty = pretty;
123    }
124
125    @Override
126    public void execute() throws MojoExecutionException, MojoFailureException {
127        if (this.skip) {
128            return;
129        }
130
131        if (!this.execFile.isFile()) {
132            getLog().debug("Not running. No execution data found.");
133            return;
134        }
135
136        try {
137            this.xmlFile.getParentFile().mkdirs();
138
139            final ReportSupport report = new ReportSupport(getLog());
140            report.loadExecutionData(this.execFile);
141            report.addXmlFormatter(this.xmlFile, "UTF-8");
142
143            final IReportVisitor visitor = report.initRootVisitor();
144            final IReportGroupVisitor group = visitor.visitGroup("XML");
145
146            processProject(report, group, this.project);
147
148            for (final MavenProject dependency : findDependencies(SCOPE_COMPILE, SCOPE_RUNTIME, SCOPE_PROVIDED)) {
149                processProject(report, group, dependency);
150            }
151
152            visitor.visitEnd();
153
154            if (this.pretty) {
155                makePretty();
156            }
157
158        } catch (final IOException e) {
159            throw new MojoExecutionException("Failed to convert to XML", e);
160        }
161    }
162
163    private void processProject(final ReportSupport report, final IReportGroupVisitor group, final MavenProject project)
164            throws IOException {
165        report.processProject(group, project.getArtifactId(), project, this.includes, this.excludes,
166                this.sourceEncoding);
167    }
168
169    private void makePretty() throws MojoExecutionException {
170        try {
171            final TransformerFactory tf = TransformerFactory.newInstance();
172
173            final Transformer transformer = tf.newTransformer();
174            transformer.setOutputProperty(OutputKeys.INDENT, "yes");
175            transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", "2");
176            transformer.setOutputProperty(OutputKeys.ENCODING, this.sourceEncoding);
177            transformer.setOutputProperty(OutputKeys.DOCTYPE_PUBLIC, "-//JACOCO//DTD Report 1.0//EN");
178            transformer.setOutputProperty(OutputKeys.DOCTYPE_SYSTEM, "report.dtd");
179
180            final Path tmp = this.xmlFile.toPath().getParent()
181                    .resolve("raw." + this.xmlFile.toPath().getFileName().toString());
182
183            Files.move(this.xmlFile.toPath(), tmp, REPLACE_EXISTING);
184
185            try (final InputStream in = new BufferedInputStream(Files.newInputStream(tmp));
186                    final OutputStream out = new BufferedOutputStream(Files.newOutputStream(this.xmlFile.toPath()));) {
187
188                final DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
189                dbf.setValidating(false);
190                dbf.setExpandEntityReferences(false);
191                dbf.setFeature("http://apache.org/xml/features/nonvalidating/load-external-dtd", false);
192                final Document doc = dbf.newDocumentBuilder().parse(in);
193
194                final DOMSource source = new DOMSource(doc);
195                final StreamResult result = new StreamResult(out);
196
197                transformer.transform(source, result);
198            }
199
200            // we only delete this if we succeeded
201            if (this.deleteRaw) {
202                Files.deleteIfExists(tmp);
203            }
204
205        } catch (final Exception e) {
206            throw new MojoExecutionException("Failed to pretty print XML file", e);
207        }
208
209    }
210
211    private List<MavenProject> findDependencies(final String... scopes) {
212        final List<MavenProject> result = new LinkedList<>();
213        final Set<String> scopeList = new HashSet<>(Arrays.asList(scopes));
214        for (final Dependency dependency : this.project.getDependencies()) {
215            if (scopeList.contains(dependency.getScope())) {
216                getLog().info("Adding dependency - " + dependency.toString());
217                final MavenProject project = findProjectFromReactor(dependency);
218                if (project != null) {
219                    getLog().info("  -> Unable to find in reactor");
220                    result.add(project);
221                }
222            }
223        }
224        return result;
225    }
226
227    private MavenProject findProjectFromReactor(final Dependency d) {
228        for (final MavenProject p : this.reactorProjects) {
229            if (p.getGroupId().equals(d.getGroupId()) && p.getArtifactId().equals(d.getArtifactId())
230                    && p.getVersion().equals(d.getVersion())) {
231                return p;
232            }
233        }
234        return null;
235    }
